语法基础创建应用 ​

flutter 中一切皆为部件或叫做组件 (widget)，下面将搭建第一个应用 Hello Flutter ！

Hello Flutter ​

dart
// material 是 flutter 的 UI 库
// link: https://material.io
import 'package:flutter/material.dart';

// main 入口函数
void main() => runApp(MyApp());

// MyApp 是顶层 App 组件类, 必须继承 StatelessWidget 类，并重写它的方法
class MyApp extends StatelessWidget {
  // @override 注解表示重写父类方法
  @override
  Widget build(BuildContext context) {
    // MaterialApp 是 material ui 的一个窗口组件
    // 这里返回了一个组件，有没有像 react 里面的 jsx
    return MaterialApp(
      // 标题
      title: 'Welcome to ---- flutter',
      // 页面的主要文件
      home: Scaffold(
        // App 顶部的 header Bar
        appBar: AppBar(
          title: Text('Welcome to Flutter')
        ),
        // APP 中间的内容
        body: Center(
          child: Text('Hello Word'),
        ),
      ),
    );
  }
}

Contain Widget 容器组件 ​

点我查看效果图

dart
import 'package:flutter/material.dart';

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title:'Text Widget',
      home:Scaffold(
        body: Center(
          child: Container(
            // 容器的内容
            child: Text('Hello Flutter', style: TextStyle(fontSize: 40.0)),
            // 容器内的对齐方式 (前面一个单词是轴向的，后面一个单词是横向的) 也可以是 center
            alignment: Alignment.topLeft,
            // 宽
            width: 700.0,
            // 高
            height: 200.0,
            // 内边距 
            // padding 必须是固定的常量值，所以要加 const 表示常量
            // EdgeInsets.all 表示上下左右 padding 10
            // EdgeInsets.fromLTRB 可以设置 “左上右下” 每个方向的内边距
            padding: const EdgeInsets.fromLTRB(10.0, 30.0, 0.0, 0.0),
            // 外边距
            margin: const EdgeInsets.all(10.0),
            // 背景颜色（纯色）
            // color: Colors.lightBlue,
            // 背景颜色（线性渐变），不能与 color 同时存在，否则会报错
            decoration: new BoxDecoration(
              // 渐变的方法
              gradient: const LinearGradient(
                colors: [Colors.lightBlue, Colors.greenAccent, Colors.purple]
              ),
              // 渐变的边框
              border: Border.all(width: 2.0, color: Colors.red)
            ),
          )
        ),
      )
    );
  }
}

ListView Widget 动态列表 ​

点我查看效果图

dart
import 'package:flutter/material.dart';

void main() => runApp(MyApp(
  items: new List<String>.generate(1000, (i) => "Item.... ${i}")
));

class MyApp extends StatelessWidget {
  // 接收 MyApp 的 items 参数，需要在类里面定义 final List<String> 类型的 items
  final List<String> items;

  // 构造函数: Key 参数语法格式是固定的，required this.items 表示构造 MyApp 实例必须传 items
  MyApp({ Key? key, required this.items }):super(key: key);

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'flutter framework....',
      home: Scaffold(
        appBar: AppBar(
          title: new Text('flutter test....'),
        ),
        // ListView.builder 动态列表
        body: new ListView.builder(
          // 列表的长度
          itemCount: items.length,
          // 每个 item 的 builder 函数, 回调函数里面有参数 context 和当前 item 的索引 index
          itemBuilder: (context, index) {
            // 返回显示的 item 组件
            return new ListTile(
              title: new Text('${items[index]}')
            );
          },
        )
      )
    );
  }
}

上面 GridView.count 的写法是比较旧的，目前 Flutter 不推荐这么写了，新的写法如下面代码：

dart
import 'package:flutter/material.dart';

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'flutter framework....',
      home: Scaffold(
        appBar: AppBar(
          title: new Text('flutter test....'),
        ),
        // GridView 网格列表
        body: GridView(
          // gridDelegate 相当于类型参数，SliverGridDelegateWithFixedCrossAxisCount 就是 GridView.count
          gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
            // 网格有几列
            crossAxisCount: 3,
            // 纵轴 item 之前的距离
            mainAxisSpacing: 2.0,
            // 横轴 item 之前的距离
            crossAxisSpacing: 2.0,
            // item 宽高比, 这个属性很重要，例如是电影封面宽高比就是 1/2 = 0.5
            childAspectRatio: 1,
          ),
          // 网格的内容
          children: <Widget>[
            new Text('Java'),
            new Text('Python'),
            new Text('Ruby'),
            new Text('Golang'),
            new Text('JavaScript'),
            new Text('Vue'),
            new Text('Dart'),
            new Text('React'),
          ],
        )
      )
    );
  }
}

dart
import 'package:flutter/material.dart';

void main() => runApp(
  MaterialApp(
    title: '导航显示01',
    home: new FirstScreen()
  )
);


class FirstScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('导航页面')),
      body: Center(
        child: RaisedButton(
          child: Text('查看商品详情页'),
          // 按钮点击事件
          onPressed: () {
            // 在堆栈里面放一个页面，参数需要传入上下文 Context 和 Flutter 提供的路由组件 MaterialPageRoute
            Navigator.push(context, MaterialPageRoute(
              builder: (context) => new SecondScreen()
            ));
          },
        ),
      ),
    );
  }
}

class SecondScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('商品详情浏览页'),),
      body: Center(
        child: RaisedButton(
          child: Text('返回'),
          onPressed: () {
            // 堆栈 pop 弹出当前上下文，就会回到上一个页面
            Navigator.pop(context);
          },
        )
      ),
    );
  }
}

导航参数传递与接收 ​

点我查看效果图

dart
import 'package:flutter/material.dart';

// 商品抽象类
class Product {
  // 商品标题
  final String title;
  // 商品描述
  final String description;
  // 构造函数
  Product(this.title, this.description);
}

void main() {
  runApp(MaterialApp(
    title: '导航的数据传递与接收',
    // 使用自定义商品列表组件，并赋值商品列表
    home: ProductList(
      products: List.generate(20, (i) => Product('技术邹: 商品$i', '这是一个商品详情,编号为: $i'))
    )
  ));
}

// 商品列表组件
class ProductList extends StatelessWidget {
  // 商品列表
  final List<Product> products;
  // 构造函数，赋值商品列表
  ProductList({ Key? key, required this.products }): super(key: key);
  // 渲染的组件
  @override
  Widget build(BuildContext context) {
    // Scaffold 页面脚手架
    return Scaffold(
      appBar: AppBar(title: Text('商品列表'),),
      body: ListView.builder(
        itemCount: products.length,
        itemBuilder: (context, index) {
          return ListTile(
            title: Text(products[index].title),
            subtitle: Text(products[index].description),
            onTap: () {
              // 点击对应商品，跳到对应的商品详情页面
              Navigator.push(
                context,
                MaterialPageRoute(
                  // 向 ProductDetail 详情页面传递一个参数 product
                  builder: (context) => ProductDetail(product: products[index])
                )
              );
            },
          );
        }
      )
    );
  }
}

// 商品详情页面
class ProductDetail extends StatelessWidget {
  final Product product;
  // 接收 Product List 页面，点击详情后，传递的 product 参数
  ProductDetail({ Key? key, required this.product }): super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('${product.title}'),),
      body: Center(
        child: Text('${product.description}'),
      ),
    );
  }
}

void main() => runApp(
  MaterialApp(
    title: '页面跳转返回参数',
    home: FirstPage(),
  )
);

class FirstPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('找小姐姐要电话'),),
      body: Center(
        child: RouteButton(),
      )
    );
  }
}

class RouteButton extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return RaisedButton(
      onPressed: () {
        _navagateToXiaoJiejie(context);
      },
      child: Text('去找小姐姐'),
    );
  }

  // 点击时触发的事件方法
  _navagateToXiaoJiejie(BuildContext context) async {
    // 异步跳转路由，await 将等待这个子页面结束，子页面回传的参数会被 result 接收 
    final result = await Navigator.push(
      context,
      MaterialPageRoute(builder: (context) => XiaoJieJie())
    );
    // 底部消息提示组件(一整段都是组件代码): Text('$result') 是显示返回的 result 内容
    Scaffold.of(context).showSnackBar(SnackBar(content: Text('$result')));
  }
}

class XiaoJieJie extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('我是小姐姐')),
      body: Center(
        child: Column(
          children: <Widget>[
            RaisedButton(
              child: Text('大长腿小姐姐'),
              onPressed: () {
                // 结束当前页面，并返回上一个页面，后面的参数是回传到上一个页面的参数，会被 result 接收
                Navigator.pop(context, '大长腿小姐姐: 110112119118');
              },
            ),
            RaisedButton(
              child: Text('大白腿小姐姐'),
              onPressed: () {
                Navigator.pop(context, '大白腿小姐姐: 110112119118');
              },
            ),
            RaisedButton(
              child: Text('黑丝小姐姐'),
              onPressed: () {
                Navigator.pop(context, '黑丝小姐姐: 110112119118');
              },
            )
          ],
        ),
      )
    );
  }
}
