main.py 主文件路由

python
from typing import Union
from fastapi import FastAPI
from services.openai_clip import OpenaiClip
from services.milvus import Milvus
from googletrans import Translator

app = FastAPI()

@app.get("/")
def read_root():
    return { "Hello": "World" }

@app.get("/clip/vector")
async def vector_embeddings(image_url: str, moment_id: str, baby_id: str, type: str, force: bool = False):
    milvus = Milvus()
    # 是否强制刷新
    if not force:
        # checkout 这个 moment 是否向量化过存储过，不重复生成向量
        moments = milvus.client.get(
            collection_name=milvus.collection_name,
            ids=[int(moment_id)],
            output_fields=["moment_id"]
        )
        if len(moments) > 0:
            return { 
                "data": { 
                    "moment_id": moment_id,
                    "image_url": image_url,
                    "msg": "Repeated generation"
                }
            }
    # 大模型转向量
    open_clip = OpenaiClip()
    image_encoding = open_clip.image_to_encoding(image_url = image_url)
    # 存储向量
    result = milvus.insert(
        embedding = image_encoding,
        moment_id = moment_id,
        baby_id = baby_id,
        type = type
    )
    return { 
        "data": { 
            "moment_id": moment_id,
            "image_url": image_url,
            "msg": "Success"
        }
    }

@app.get("/clip/vector/search")
async def vector_search(search_text: str, baby_id: str, type: str, page: int, limit: int):
    # 翻译文案为英文
    translator = Translator()
    translation = translator.translate(search_text, 'en')
    translation_search_text = translation.text
    # 将文案转向量
    open_clip = OpenaiClip()
    text_encoding = open_clip.text_to_encoding(text = translation_search_text)
    # 查询
    milvus = Milvus()
    result = milvus.search(
        embedding = text_encoding,
        baby_id = baby_id,
        type = type,
        page = page,
        limit = limit,
    )
    return result


if __name__ == '__main__':
    import uvicorn
    uvicorn.run("main:app", host="0.0.0.0", port=8080, workers=5, log_config='./uvicorn_config.json')

services/openai_clip.py 大模型 CLIP 处理向量

python
import os
import requests
import torch
from PIL import Image
from transformers import CLIPProcessor, CLIPModel, AutoTokenizer
from pymilvus import MilvusClient

device = "cuda" if torch.cuda.is_available() else "cpu"

processor3 = CLIPProcessor.from_pretrained("openai/clip-vit-base-patch32", use_fast=True)
model3 = CLIPModel.from_pretrained("openai/clip-vit-base-patch32").to(device)
tokenizer3 = AutoTokenizer.from_pretrained("openai/clip-vit-base-patch32")

class OpenaiClip:
    def image_to_encoding(self, image_url = str):
        image = Image.open(requests.get(image_url, stream=True).raw).convert('RGB')
        inputs = processor3(images=image, return_tensors="pt").to(device)
        image_features = model3.get_image_features(**inputs)
        image_feature = image_features[0].detach().cpu().numpy()
        return image_feature.tolist()

    def text_to_encoding(self, text = str):
        inputs = tokenizer3([text], padding=True, return_tensors="pt").to(device)
        text_features = model3.get_text_features(**inputs)
        text_feature = text_features[0].detach().cpu().numpy()
        return text_feature.tolist()

services/milvus.py 向量数据库，数据入库与查询

python
from pymilvus import connections, db, MilvusClient, DataType

db_name = "timehue_moment_vector_database"
conn = connections.connect(
    host="127.0.0.1",
    port="19530",
)
db.using_database(db_name)
print("databases: ", db.list_database())


class Milvus:
    def __init__(self, collection_name = "vector_db_1", dimension = 512):
        self.client = MilvusClient(uri="http://127.0.0.1:19530", token="root:Milvus")
        self.collection_name = collection_name

        if not self.client.has_collection(collection_name = collection_name):
            # 添加字段
            schema = MilvusClient.create_schema(auto_id=False, enable_dynamic_field=True)
            schema.add_field(field_name="moment_id", datatype=DataType.INT64, is_primary=True, auto_id=False, description="moment_id")
            schema.add_field(field_name="vector", datatype=DataType.FLOAT_VECTOR, dim=dimension, description="vector")
            schema.add_field(field_name="baby_id", datatype=DataType.INT32, description="baby_id")
            schema.add_field(field_name="type", datatype=DataType.VARCHAR, max_length=60, description="type")
            # 添加索引
            index_params = self.client.prepare_index_params()
            index_params.add_index(field_name="moment_id", index_type="STL_SORT")
            index_params.add_index(field_name="baby_id", index_type="STL_SORT")
            index_params.add_index(field_name="vector", index_type="AUTOINDEX", metric_type="COSINE")
            # 创建集合
            self.client.create_collection(
                collection_name="vector_db_1",
                schema=schema,
                index_params=index_params
            )
            print(f"{db_name}: ", self.client.list_collections())

    def insert(self, embedding, moment_id, baby_id, type):
        res = self.client.insert(
            self.collection_name,
            data=[
                {
                    "vector": embedding, 
                    "moment_id": int(moment_id),
                    "baby_id": int(baby_id),
                    "type": type
                }
            ] 
        )
        return res

    def search(self, embedding, baby_id, type = None, page: int = 1, limit: int = 10, max_search_length: int = 40):
        filter = f"baby_id == {baby_id} and type == '{type}'" if type else f"baby_id == {baby_id}"

        # 分页搜索
        page_res = self.client.search(
            self.collection_name,
            data=[embedding],
            filter=filter,
            output_fields=["moment_id", "baby_id", "type"],
            search_params={
                "metric_type": "COSINE",
                "offset": (page - 1) * limit
            },
            limit=limit,
        )
        
        total = 0
        # 如果需要分页，则需要 total 判断是否有下一页
        if max_search_length > limit:
            iterator = self.client.query_iterator(
                self.collection_name,
                data=[embedding],
                filter=filter,
                output_fields=["moment_id", "baby_id", "type"],
                search_params={
                    "metric_type": "COSINE"
                },
                batch_size=max_search_length
            )
            while True:
                result = iterator.next()
                total += len(result)
                if not result or total >= max_search_length:
                    iterator.close()
                    break
        # 不需要分页，直接返回数组长度
        else:
            total = len(page_res[0])

        return {
            "total": total,
            "data": page_res[0],
            "next_page": total > (page * limit),
        }
